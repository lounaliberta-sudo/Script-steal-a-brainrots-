def obfuscate(text):
    result = ""
    for char in text:
        if char.isalpha():
            # Shift character by 1
            if char.lower() == 'z':
                shifted = 'a'
            else:
                shifted = chr(ord(char) + 1)
            # Preserve case
            if char.isupper():
                result += shifted.upper()
            else:
                result += shifted.lower()
        else:
            result += char
    return result
